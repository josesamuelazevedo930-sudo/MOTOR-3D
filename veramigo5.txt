if(document.getElementById('ui-multi')) document.getElementById('ui-multi').remove();

const mqttScript = document.createElement('script');
mqttScript.src = 'https://cdn.jsdelivr.net/npm/mqtt@4.1.0/dist/mqtt.min.js';
document.head.appendChild(mqttScript);

mqttScript.onload = () => { setupV18(); };

function setupV18() {
    const ui = document.createElement('div');
    ui.id = 'ui-multi';
    ui.style.cssText = "position:absolute; top:10px; left:10px; z-index:10000; font-family:sans-serif; background:rgba(0,0,0,0.95); padding:10px; border-radius:10px; border:2px solid #00ffff; color:white; width:220px;";
    ui.innerHTML = `
        <b style="color:#00ffff;">SIRI V18 - FORCE SKIN</b><br>
        <div id="setup-mp">
            <input id="mp-room" placeholder="Nº da Sala" style="width:90%; margin:5px 0;">
            <p style="font-size:10px; margin:2px 0;">Nome da Skin Local (ex: bob.glb):</p>
            <input id="mp-skin-file" placeholder="bob.glb" style="width:90%; margin:5px 0;">
            <button id="btn-con" style="width:100%; padding:8px; background:green; color:white; border:none; font-weight:bold;">CONECTAR</button>
        </div>
        <div id="chat-area" style="display:none; margin-top:5px;">
            <div id="status-render" style="font-size:10px; color:#ff0; margin-bottom:5px;">Status: Buscando Cena...</div>
            <div id="chat-logs" style="height:80px; overflow-y:auto; font-size:10px; background:#000; color:#0f0; padding:5px;"></div>
            <input type="text" id="chat-input" placeholder="Oi!" style="width:65%;">
            <button id="btn-send" style="width:30%;">OK</button>
        </div>
    `;
    document.body.appendChild(ui);

    let client, sala, skinLocal, meuID = "TV-" + Math.floor(Math.random()*999);
    let remotePlayers = {}; 

    document.getElementById('btn-con').onclick = () => {
        sala = document.getElementById('mp-room').value || "1";
        skinLocal = document.getElementById('mp-skin-file').value || "bob.glb"; // Aqui você escolhe a skin!
        
        client = mqtt.connect('wss://broker.emqx.io:8084/mqtt', { clientId: meuID });

        client.on('connect', () => {
            document.getElementById('setup-mp').style.display = 'none';
            document.getElementById('chat-area').style.display = 'block';
            client.subscribe('siri/multi/' + sala);
            addLog("SISTEMA", "Conectado!");
        });

        client.on('message', (topic, msg) => {
            try {
                const data = JSON.parse(msg.toString());
                if (data.id !== meuID) handleData(data);
            } catch (e) {}
        });
    };

    function handleData(data) {
        if (data.type === 'chat') { addLog(data.id, data.text); return; }

        // TENTA ACHAR A CENA DE TODAS AS FORMAS POSSÍVEIS
        const world = window.scene || (window.game && window.game.scene) || (window.world && window.world.scene);
        
        if (!world) {
            document.getElementById('status-render').innerText = "⚠️ ERRO: Cena não encontrada!";
            return;
        }

        if (!remotePlayers[data.id]) {
            document.getElementById('status-render').innerText = "✅ Renderizando amigo...";
            
            // 1. Cria um Cubo temporário só para marcar a posição
            const geo = new THREE.BoxGeometry(2, 5, 2);
            const mat = new THREE.MeshBasicMaterial({ color: 0x00ffff });
            remotePlayers[data.id] = new THREE.Mesh(geo, mat);
            world.add(remotePlayers[data.id]);

            // 2. Tenta carregar a SKIN QUE VOCÊ ESCOLHEU do seu USB
            if (window.loaderGLTF) {
                window.loaderGLTF.load(skinLocal, (gltf) => {
                    world.remove(remotePlayers[data.id]);
                    remotePlayers[data.id] = gltf.scene;
                    world.add(remotePlayers[data.id]);
                    addLog("SISTEMA", "Boneco carregado!");
                }, undefined, (err) => {
                    addLog("AVISO", "Não achei " + skinLocal + ", usando cubo.");
                });
            }
        }

        // Atualizar posição
        if (remotePlayers[data.id]) {
            remotePlayers[data.id].position.set(parseFloat(data.x), parseFloat(data.y), parseFloat(data.z));
            remotePlayers[data.id].rotation.y = parseFloat(data.ry);
        }
    }

    setInterval(() => {
        if (client && client.connected && window.player) {
            client.publish('siri/multi/' + sala, JSON.stringify({
                type: 'sync', id: meuID,
                x: player.position.x, y: player.position.y, z: player.position.z, ry: player.rotation.y
            }));
        }
    }, 200);

    function addLog(nick, msg) {
        const logs = document.getElementById('chat-logs');
        logs.innerHTML += `<div><b>${nick}:</b> ${msg}</div>`;
        logs.scrollTop = logs.scrollHeight;
    }

    document.getElementById('btn-send').onclick = () => {
        const inp = document.getElementById('chat-input');
        if (inp.value && client) {
            client.publish('siri/multi/' + sala, JSON.stringify({ type: 'chat', id: meuID, text: inp.value }));
            addLog("Eu", inp.value);
            inp.value = "";
        }
    };
}
