// --- SISTEMA DE NPCs COM VIDA PRÓPRIA ---
(function() {
    const npcUrl = "https://threejs.org/examples/models/gltf/Soldier.glb"; // Pode mudar para qualquer .glb
    const npcs = [];
    const npcCount = 3;

    // Elemento HTML pai para as barras de vida
    const uiContainer = document.createElement('div');
    uiContainer.id = 'npc-ui-container';
    uiContainer.style.position = 'absolute';
    uiContainer.style.top = '0';
    uiContainer.style.left = '0';
    uiContainer.style.width = '100%';
    uiContainer.style.height = '100%';
    uiContainer.style.pointerEvents = 'none'; // Não atrapalha os cliques no jogo
    uiContainer.style.zIndex = '99';
    document.body.appendChild(uiContainer);

    // Função para criar a barra de vida flutuante no HTML
    function createLifeBar() {
        const barContainer = document.createElement('div');
        barContainer.style.position = 'absolute';
        barContainer.style.width = '60px';
        barContainer.style.height = '8px';
        barContainer.style.background = '#333';
        barContainer.style.border = '1px solid #fff';
        barContainer.style.borderRadius = '4px';
        barContainer.style.display = 'none';

        const barFill = document.createElement('div');
        barFill.style.width = '100%';
        barFill.style.height = '100%';
        barFill.style.background = '#00ff00';
        barFill.style.borderRadius = '3px';

        barContainer.appendChild(barFill);
        uiContainer.appendChild(barContainer);

        return { container: barContainer, fill: barFill };
    }

    // Carregar os NPCs usando o loader original do seu motor
    for (let i = 0; i < npcCount; i++) {
        loaderGLTF.load(npcUrl, (gltf) => {
            const npcMesh = gltf.scene.clone(); // Clona para criar cópias independentes
            
            // Posiciona os NPCs espalhados pelo mapa
            let posX = (Math.random() - 0.5) * 300 + 50;
            let posZ = (Math.random() - 0.5) * 300 + 50;
            npcMesh.position.set(posX, 0, posZ);

            // Ajusta a escala baseado no tamanho do soldado
            npcMesh.scale.set(7, 7, 7);
            scene.add(npcMesh);

            // Configura as animações individuais do NPC
            let npcMixer = new THREE.AnimationMixer(npcMesh);
            let walk = npcMixer.clipAction(gltf.animations.find(a => a.name.toLowerCase().match(/run|walk|move/)) || gltf.animations[0]);
            walk.play();

            // Dados lógicos de "vida própria" do NPC
            const npcData = {
                mesh: npcMesh,
                mixer: npcMixer,
                health: 100,
                maxHealth: 100,
                ui: createLifeBar(),
                targetPos: new THREE.Vector3(posX, 0, posZ),
                speed: 0.3 + Math.random() * 0.3,
                changeTargetTimer: 0
            };

            npcs.push(npcData);
            console.log(`NPC ${i+1} vivo no mapa!`);
        });
    }

    // Intercepta o laço de animação principal do seu jogo
    const originalAnimate = window.animate;
    const tempV3 = new THREE.Vector3();

    window.animate = function() {
        originalAnimate(); // Roda a física e render do seu jogo primeiro
        
        let delta = clock.getDelta(); // Usa o clock do seu motor

        npcs.forEach(npc => {
            // 1. Atualizar animação 3D do NPC
            if (npc.mixer) npc.mixer.update(0.015); // Força um delta estável para o mixer interno

            // 2. Inteligência Artificial: Escolha de caminhos aleatórios
            npc.changeTargetTimer -= 0.016;
            if (npc.changeTargetTimer <= 0) {
                // Escolhe um novo ponto no mapa para caminhar a cada 5-10 segundos
                npc.targetPos.x = (Math.random() - 0.5) * 400;
                npc.targetPos.z = (Math.random() - 0.5) * 400;
                npc.changeTargetTimer = 5 + Math.random() * 5;
            }

            // 3. Movimentação física em direção ao objetivo
            let dir = new THREE.Vector3().subVectors(npc.targetPos, npc.mesh.position);
            dir.y = 0;
            if (dir.length() > 2) {
                dir.normalize();
                npc.mesh.position.addScaledVector(dir, npc.speed);
                // Rotaciona o NPC para olhar para onde está andando
                npc.mesh.rotation.y = Math.atan2(-dir.x, -dir.z);
            }

            // 4. Lógica de Interação: Se o jogador chegar muito perto, o NPC te olha
            if (player) {
                let distAoJogador = npc.mesh.position.distanceTo(player.position);
                if (distAoJogador < 30) {
                    // Para e encara o jogador se ele invadir o espaço dele
                    npc.mesh.lookAt(player.position.x, npc.mesh.position.y, player.position.z);
                    
                    // Simulação de perda de vida opcional: NPCs se desgastam perto de você
                    npc.health -= 0.05; 
                    if(npc.health < 0) npc.health = npc.maxHealth; // Ressuscita para não sumir
                }
            }

            // 5. Projeção 3D para 2D (Fazer a barra de vida seguir a cabeça do NPC na tela)
            npc.mesh.getWorldPosition(tempV3);
            tempV3.y += 12; // Altura da barra acima da cabeça do modelo
            tempV3.project(camera); // Projeta a coordenada 3D na câmera do jogo

            // Verifica se o NPC está na frente da câmera (visível)
            if (tempV3.z <= 1) {
                const x = (tempV3.x * .5 + .5) * window.innerWidth;
                const y = (tempV3.y * -.5 + .5) * window.innerHeight;

                npc.ui.container.style.display = 'block';
                npc.ui.container.style.left = `${x - 30}px`; // Centraliza os 60px de largura
                npc.ui.container.style.top = `${y}px`;
                
                // Atualiza a largura da barra verde com base na porcentagem de vida
                const pct = (npc.health / npc.maxHealth) * 100;
                npc.ui.fill.style.width = `${Math.max(0, pct)}%`;
            } else {
                npc.ui.container.style.display = 'none';
            }
        });
    };
})();
