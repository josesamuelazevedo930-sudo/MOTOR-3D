import os
import subprocess
import curses

# Caminho padrão das músicas no Android (Pasta Download ou Music)
# Altere para os locais onde suas músicas estão guardadas
PASTAS_MUSICA = [
    "/sdcard/Download",
    "/sdcard/Music",
    "/sdcard/Movies"
]

def buscar_musicas():
    lista_musicas = []
    extensoes = ('.mp3', '.m4a', '.wav', '.flac')
    
    for pasta in PASTAS_MUSICA:
        if os.path.exists(pasta):
            for raiz, _, arquivos in os.walk(pasta):
                for arquivo in arquivos:
                    if arquivo.lower().endswith(extensoes):
                        lista_musicas.append(os.path.join(raiz, arquivo))
    return sorted(lista_musicas)

def main(stdscr):
    # Configurações iniciais da interface no terminal
    curses.curs_set(0)
    stdscr.nodelay(1)
    stdscr.timeout(100)
    
    musicas = buscar_musicas()
    if not musicas:
        stdscr.addstr(0, 0, "Nenhuma música encontrada nas pastas padrões do celular!")
        stdscr.addstr(2, 0, "Aperte qualquer tecla para sair...")
        stdscr.nodelay(0)
        stdscr.getch()
        return

    selecionado = 0
    processo_audio = None
    musica_atual = ""

    while True:
        stdscr.clear()
        stdscr.addstr(0, 0, "=== TERMUX MP3 PLAYER ===", curses.A_BOLD)
        stdscr.addstr(1, 0, "Use as setas [↑ / ↓] para navegar | [Enter] para tocar | [Q] para Sair")
        stdscr.addstr(2, 0, "---------------------------------------------------------------------")

        # Exibir a lista de músicas encontradas
        for idx, caminho in enumerate(musicas):
            nome_arquivo = os.path.basename(caminho)
            if idx == selecionado:
                stdscr.addstr(4 + idx, 0, f"> {nome_arquivo}", curses.A_REVERSE)
            else:
                stdscr.addstr(4 + idx, 0, f"  {nome_arquivo}")

        stdscr.addstr(curses.LINES - 2, 0, f"Tocando agora: {musica_atual if musica_atual else 'Nenhuma'}", curses.A_DIM)

        # Capturar entrada de teclado
        key = stdscr.getch()

        if key == ord('q') or key == ord('Q'):
            if processo_audio:
                processo_audio.terminate()
            break
        elif key == curses.KEY_UP:
            selecionado = (selecionado - 1) % len(musicas)
        elif key == curses.KEY_DOWN:
            selecionado = (selecionado + 1) % len(musicas)
        elif key == 10 or key == 13: # Tecla Enter
            if processo_audio:
                processo_audio.terminate()
            
            caminho_musica = musicas[selecionado]
            musica_atual = os.path.basename(caminho_musica)
            
            # Executa o mpv em segundo plano enviando o áudio para o OpenSL do Android
            processo_audio = subprocess.Popen(
                ['mpv', '--ao=opensl', '--no-video', caminho_musica],
                stdout=subprocess.DEVNULL,
                stderr=subprocess.DEVNULL
            )

if __name__ == "__main__":
    curses.wrapper(main)
