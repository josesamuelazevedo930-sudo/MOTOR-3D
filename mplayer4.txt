if(document.getElementById('ui-multi')) document.getElementById('ui-multi').remove();

const mqttScript = document.createElement('script');
mqttScript.src = 'https://cdn.jsdelivr.net/npm/mqtt@4.1.0/dist/mqtt.min.js';
document.head.appendChild(mqttScript);

mqttScript.onload = () => { setupMultiplayerV6(); };

function setupMultiplayerV6() {
    // BOTÃƒO M-PLAYER
    const btnMenu = document.createElement('button');
    btnMenu.innerHTML = "M-PLAYER";
    btnMenu.style.cssText = "position:absolute; right:20px; top:240px; width:110px; padding:12px; color:white; border:none; font-weight:bold; border-radius:8px; z-index:1001; background:#ff00ff; box-shadow:0 4px #000; cursor:pointer;";
    btnMenu.onclick = () => {
        const p = document.getElementById('ui-multi');
        p.style.display = p.style.display === 'none' ? 'block' : 'none';
    };
    document.body.appendChild(btnMenu);

    // LOG DE CHAT (VERDE - CANTO SUPERIOR ESQUERDO)
    const chatLog = document.createElement('div');
    chatLog.id = 'chat-visual-log';
    chatLog.style.cssText = "position:absolute; left:10px; top:10px; width:220px; height:150px; overflow-y:auto; background:rgba(0,0,0,0.5); color:#00ff00; font-family:monospace; font-size:12px; padding:5px; z-index:1000; border-left:3px solid #00ff00; pointer-events:none;";
    document.body.appendChild(chatLog);

    // PAINEL DE CONTROLE
    const ui = document.createElement('div');
    ui.id = 'ui-multi';
    ui.style.cssText = "position:absolute; right:20px; top:295px; width:250px; background:rgba(0,0,0,0.95); border-radius:12px; padding:15px; color:white; border:1px solid #ff00ff; z-index:1000; display:none; font-family:sans-serif;";
    ui.innerHTML = `
        <div id="setup-mp">
            <input id="mp-name" placeholder="Nome" style="width:90%; padding:8px; margin-bottom:5px; background:#222; color:#fff; border:1px solid #444;">
            <input id="mp-room" placeholder="Sala" value="1" style="width:90%; padding:8px; margin-bottom:5px; background:#222; color:#fff; border:1px solid #444;">
            <div style="font-size:10px; margin-bottom:5px;">COR DO NICK:</div>
            <input type="color" id="mp-color" value="#ff00ff" style="width:100%; height:30px; margin-bottom:10px; border:none; background:none;">
            <button id="btn-con" style="width:100%; padding:10px; background:#ff00ff; color:black; font-weight:bold; border-radius:6px;">CONECTAR</button>
        </div>
        <div id="mp-active" style="display:none;">
            <div style="display:flex; gap:5px;">
                <input id="chat-in" placeholder="Diga algo..." style="flex:1; background:#222; color:#fff; border:1px solid #444; padding:5px;">
                <button id="btn-send" style="padding:5px 10px; background:#0ff; color:#000; font-weight:bold; border:none;">OK</button>
            </div>
            <button id="btn-exit" style="width:100%; margin-top:10px; background:#ff3333; color:white; border:none; padding:8px; border-radius:4px; font-weight:bold;">SAIR DA SALA</button>
            <div id="st-count" style="font-size:9px; color:#aaa; margin-top:8px; text-align:center;">ðŸ‘¥ JOGADORES: 0</div>
        </div>
    `;
    document.body.appendChild(ui);

    let client, sala, meuNome, minhaCor, remotePlayers = {};
    const jScene = window.scene || scene;

    function mostrarBalonete(playerObj, texto) {
        if (playerObj.msgSprite) playerObj.remove(playerObj.msgSprite);
        const canvas = document.createElement('canvas');
        const ctx = canvas.getContext('2d');
        canvas.width = 512; canvas.height = 128;
        ctx.fillStyle = "rgba(0,0,0,0.6)"; ctx.fillRect(0, 0, 512, 128);
        ctx.fillStyle = 'white'; ctx.font = 'bold 40px Arial'; ctx.textAlign = 'center';
        ctx.fillText(texto, 256, 75);
        const sprite = new THREE.Sprite(new THREE.SpriteMaterial({ map: new THREE.CanvasTexture(canvas), depthTest: false }));
        sprite.scale.set(12, 3, 1);
        sprite.position.y = 15; // Acima do nick
        playerObj.add(sprite);
        playerObj.msgSprite = sprite;
        setTimeout(() => { if(playerObj.msgSprite === sprite) playerObj.remove(sprite); }, 6000);
    }

    function criarAvatarRemoto(id, cor) {
        const grupo = new THREE.Group();
        loaderGLTF.load('https://threejs.org/examples/models/gltf/Soldier.glb', (gltf) => {
            const model = gltf.scene;
            model.scale.set(7, 7, 7);
            grupo.add(model);
            grupo.mixer = new THREE.AnimationMixer(model);
            grupo.anims = {};
            gltf.animations.forEach(a => { grupo.anims[a.name.toLowerCase()] = grupo.mixer.clipAction(a); });
            if(grupo.anims.idle) grupo.anims.idle.play();
        });

        const canvasN = document.createElement('canvas');
        const ctxN = canvasN.getContext('2d');
        canvasN.width = 256; canvasN.height = 64;
        ctxN.fillStyle = cor; ctxN.font = 'bold 30px Arial'; ctxN.textAlign = 'center';
        ctxN.fillText(id, 128, 45);
        const labelN = new THREE.Sprite(new THREE.SpriteMaterial({ map: new THREE.CanvasTexture(canvasN), depthTest: false }));
        labelN.scale.set(6, 1.5, 1); labelN.position.y = 10;
        grupo.add(labelN);
        jScene.add(grupo);
        return grupo;
    }

    document.getElementById('btn-con').onclick = () => {
        meuNome = document.getElementById('mp-name').value || "User" + Math.floor(Math.random()*99);
        sala = document.getElementById('mp-room').value || "1";
        minhaCor = document.getElementById('mp-color').value;
        document.getElementById('setup-mp').style.display = 'none';
        document.getElementById('mp-active').style.display = 'block';

        client = mqtt.connect('wss://broker.emqx.io:8084/mqtt', { clientId: 'siri_' + Math.random().toString(36).substr(2, 5) });

        client.on('message', (topic, msg) => {
            try {
                const data = JSON.parse(msg.toString());
                if (data.id === meuNome) return;

                if (data.type === 'sync') {
                    if (!remotePlayers[data.id]) {
                        remotePlayers[data.id] = criarAvatarRemoto(data.id, data.cor || '#ff00ff');
                        updateCounter();
                    }
                    const rp = remotePlayers[data.id];
                    rp.position.set(data.x, data.y, data.z);
                    rp.rotation.y = data.ry;
                    rp.lastSeen = Date.now();
                    
                    if(rp.mixer && data.anim) {
                        const target = rp.anims[data.anim.toLowerCase()];
                        if(target && !target.isRunning()) {
                            Object.values(rp.anims).forEach(a => a.stop());
                            target.play();
                        }
                    }
                }
                
                if (data.type === 'chat') {
                    addLog(data.id, data.text);
                    if (remotePlayers[data.id]) mostrarBalonete(remotePlayers[data.id], data.text);
                }
            } catch(e){}
        });
        client.on('connect', () => { client.subscribe('siri/multi/' + sala); addLog("SISTEMA", "Conectado na Sala " + sala); });
    };

    document.getElementById('btn-exit').onclick = () => { location.reload(); };

    setInterval(() => {
        const pLocal = window.player || player;
        if (client && client.connected && pLocal) {
            let curAnim = (window.activeAction && window.activeAction._clip) ? window.activeAction._clip.name : "idle";
            client.publish('siri/multi/' + sala, JSON.stringify({
                type: 'sync', id: meuNome, cor: minhaCor,
                x: pLocal.position.x, y: pLocal.position.y, z: pLocal.position.z,
                ry: pLocal.rotation.y, anim: curAnim
            }));
        }
        for(let id in remotePlayers) {
            if(remotePlayers[id].mixer) remotePlayers[id].mixer.update(0.1);
            if(Date.now() - remotePlayers[id].lastSeen > 5000) { jScene.remove(remotePlayers[id]); delete remotePlayers[id]; updateCounter(); }
        }
    }, 100);

    function addLog(n, t) {
        const logs = document.getElementById('chat-visual-log');
        logs.innerHTML += `<div><b>${n}:</b> ${t}</div>`;
        logs.scrollTop = logs.scrollHeight;
    }

    function updateCounter() { document.getElementById('st-count').innerText = "ðŸ‘¥ JOGADORES: " + Object.keys(remotePlayers).length; }

    document.getElementById('btn-send').onclick = () => {
        const i = document.getElementById('chat-in');
        if(i.value && client) {
            client.publish('siri/multi/' + sala, JSON.stringify({ type: 'chat', id: meuNome, text: i.value }));
            addLog("Eu", i.value);
            i.value = "";
        }
    };
}
